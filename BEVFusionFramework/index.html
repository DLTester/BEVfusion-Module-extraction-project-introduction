BEVFusion Framework（BEVFusion 总体框架）¶

BEVFusionFramework 的作用是：
统一组织图像侧、点云侧与 BEV 空间融合模块，
实现 BEVFusion 的完整推理流程，包括图像处理、点云处理、BEV 生成与最终融合。

本框架包含四类核心功能：

  1. 图像侧处理流程
    调用 Camera Encoder Interface，对多视角图像提取语义与深度特征，并生成 Camera BEV 特征。

  2. 点云侧处理流程
    调用 Lidar Preprocess Interface + 点云主干网络，生成 LiDAR BEV 特征。

  3. BEV 空间融合流程
    将图像 BEV 与点云 BEV 特征进行拼接或其他融合方式，形成统一 BEV 表示。

  4. 推理组织逻辑
    封装输入 → 提取特征 → BEV生成 → 融合 的完整流程，作为模型整体入口。

class BEVFusionFramework(nn.Module):
    """
    BEVFusionFramework（BEVfusion 算子级融合框架）。

    该类把算子层 + 网络层串联起来，实现：
    多视角图像 + LiDAR 点云 → Fused BEV Features 的完整前向流程。
    """

    def __init__(
        self,
        camera_encoder: CameraEncoderInterface,
        lidar_preprocess: LidarPreprocessInterface,
        camera_fpn_cfg: dict,
        depth_head_cfg: dict,
        voxel_cfg: dict,
        vfe_cfg: dict,
        backbone3d_cfg: dict,
        bev_encoder_cfg: dict,
    ):
        super().__init__()  # 调用父类构造

        # 接口模块
        self.camera_encoder = camera_encoder              # 相机编码接口
        self.lidar_preprocess = lidar_preprocess          # 点云预处理接口

        # Camera side 网络层（算子 1 & 2）
        self.camera_fpn = CameraFPNFusion(**camera_fpn_cfg)      # 图像多尺度特征融合
        self.depth_head = DepthDistributionHead(**depth_head_cfg)  # 像素深度分布预测

        # LiDAR side 网络层（算子 7 / 8 / 9）
        self.lidar_voxelization = LidarVoxelization(**voxel_cfg)     # 点云体素化
        self.vfe = VoxelFeatureEncoder(**vfe_cfg)                    # 体素特征编码
        self.backbone3d = Sparse3DBackbone(**backbone3d_cfg)         # 稀疏 3D 主干

        # BEV 融合编码（算子 12）
        self.bev_encoder = BEVFusionEncoder(**bev_encoder_cfg)       # BEV 融合编码器

    def forward(
        self,
        images: torch.Tensor,
        raw_points,
        pixels_2d: torch.Tensor,
        depths: torch.Tensor,
        K: torch.Tensor,
        R: torch.Tensor,
        t: torch.Tensor,
        bev_indices: torch.Tensor,
        batch_size: int,
    ) -> torch.Tensor:
        """
        Args:
            images (torch.Tensor):
                多视角图像 (B, N_cam, 3, H, W)。                 # 图像输入
            raw_points:
                原始点云数据，由 LidarPreprocessInterface 负责解析。 # 点云输入
            pixels_2d (torch.Tensor):
                像素网格 (H_f, W_f, 2)，用于几何升维。            # 像素坐标
            depths (torch.Tensor):
                深度离散 bins (D,)。                              # 深度取样
            K (torch.Tensor):
                相机内参 (3, 3)。                                 # 内参
            R (torch.Tensor):
                相机到目标坐标系旋转矩阵 (3, 3)。                 # 旋转
            t (torch.Tensor):
                相机到目标坐标系平移向量 (3,)。                   # 平移
            bev_indices (torch.Tensor):
                BEV 索引 (B, D, H_f, W_f, 2)。                    # BEV 网格索引
            batch_size (int):
                当前 batch 大小。                                 # B

        Returns:
            torch.Tensor:
                Fused BEV 特征 (B, C_bev, H_bev, W_bev)。         # 输出 BEV
        """
        # ===================== Camera side =====================

        # 接口：多视角图像预处理 + backbone → 多尺度特征列表
        multi_scale_feats = self.camera_encoder(images)                 # [ (B,C_i,H_i,W_i), ... ]

        # 算子 1：FPN 多尺度特征融合
        cam_fpn_feat = self.camera_fpn(multi_scale_feats)               # (B, C_fpn, H_f, W_f)

        # 算子 2：像素深度分布预测
        depth_prob = self.depth_head(cam_fpn_feat)                      # (B, D, H_f, W_f)

        # 算子 5：深度加权特征展开（调用自定义算子）
        feat_3d = torch.ops.bevfusion_ops.depth_weighted_feature_expand(
            cam_fpn_feat,
            depth_prob,
        )                                                                # (B, D, C_fpn, H_f, W_f)

        # 算子 3：像素 2D→3D 升维（可用于几何可视化 / 前置计算）
        points_cam = torch.ops.bevfusion_ops.camera_lift_2d_to_3d(
            pixels_2d,
            depths,
            K,
            R,
            t,
        )                                                                # (D, H_f, W_f, 3)

        # 算子 4：坐标变换到目标坐标系（如 LiDAR / Ego）
        points_world = torch.ops.bevfusion_ops.camera_geometry_transform(
            points_3d=points_cam,
            R=R,
            t=t,
        )                                                                # (D, H_f, W_f, 3)
        _ = points_world                                                 # 这里示意使用，可在需要时进一步利用

        # 算子 6：Camera BEV 聚合
        B, D, C_fpn, H_f, W_f = feat_3d.shape                            # 读取尺寸
        bev_h = bev_indices.shape[2]                                     # BEV 高度
        bev_w = bev_indices.shape[3]                                     # BEV 宽度

        cam_bev = torch.ops.bevfusion_ops.camera_bev_pooling(
            feat_3d,
            bev_indices,
            bev_h,
            bev_w,
        )                                                                # (B, C_fpn, bev_h, bev_w)

        # ===================== LiDAR side ======================

        # 接口：点云预处理
        points = self.lidar_preprocess(raw_points)                       # (B, N, C_pts)

        # 算子 7：体素化
        voxels, coords, num_points = self.lidar_voxelization(points)    # (M,T,Cv),(M,4),(M,)

        # 算子 8：体素特征编码
        voxel_feats = self.vfe(voxels, num_points)                      # (M, C_vfe)

        # 算子 9：3D 主干网络
        feat_3d_lidar = self.backbone3d(
            voxel_feats,
            coords,
            batch_size,
        )                                                                # (B, C_3d, Z, Y, X)

        # 算子 10：高度维压缩 → LiDAR BEV
        lidar_bev = torch.ops.bevfusion_ops.lidar_flatten_z_to_bev(
            feat_3d_lidar,
        )                                                                # (B, C_lidar, bev_h, bev_w)

        # ===================== BEV 融合 ========================

        # 算子 11：Camera BEV + LiDAR BEV 通道拼接
        bev_concat = torch.ops.bevfusion_ops.bev_concat_fusion(
            cam_bev,
            lidar_bev,
        )                                                                # (B, C_in, bev_h, bev_w)

        # 算子 12：BEV 融合编码
        bev_out = self.bev_encoder(bev_concat)                          # (B, C_bev, bev_h, bev_w)

        return bev_out                                                  # 返回最终 Fused BEV Features
